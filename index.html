                        function calculateRating($orders, $expert) {
                          // initial value
                          $rating = 0;
                          $totalWeight = 0;
                          // new user limitations
                          $newExpertLimitations = [
                            "canAttachFiles" => false,
                            "maxConcurrentOrders" => 4,
                            "hasForumAccess" => false,
                          ];
                          // existing user limitations
                          if ($expert['orders_completed'] > 100) {
                            $newExpertLimitations['canAttachFiles'] = true;
                            if ($expert['orders_completed'] > 200) {
                              $newExpertLimitations['maxConcurrentOrders'] = null;
                              $newExpertLimitations['hasForumAccess'] = true;
                            }
                          }
                          // order processing
                          foreach ($orders as $i => $order) {
                            // base rating
                            $xi = $order['rating'] ?? 5;
                          // penalties
                          if ($order['delivered_late']) $xi -= 1;
                          if (!$order['confirmed_at_50_percent']) $xi = 2;
                          if ($order['redistributed']) {
                            if ($order['redistribution'] == 100) $xi = 2;
                            elseif ($order['redistribution'] == 50) $xi = 3; // Assuming, that 50/50 gives rating 3
                            else $xi = 4;
                          }
                          // order value calculation
                          $alpha_i = 0;
                          if ($order['price'] < 100) $alpha_i = 0.1;
                          elseif ($order['price'] >= 100 && $order['price'] < 1000) $alpha_i = 0.9;
                          elseif ($order['price'] >= 5000) $alpha_i = 1.5;
                          // additional influencing factors
                          if (!$order['user_rating']) $alpha_i = 0;
                          if ($order['premium']) $alpha_i *= 2;
                          // time coefficient
                          if ($i > 40 && $i <= 100) $alpha_i *= 1 - 0.005 * ($i - 40);
                          elseif ($i > 100) $alpha_i *= 0.5 - 0.005 * ($i - 100);
                          // overal rating calculation
                          $rating += $xi * $alpha_i;
                          $totalWeight += $alpha_i;
                          // additional coeffs
                          if ($expert['order_refusal_rate'] > 0.3) $totalWeight -= 0.5;
                          if ($expert['order_confirmation_rate'] > 0.95) $totalWeight += 0.1;
                          if ($expert['redistribution_rate'] > 0.2) $totalWeight -= quote("function calculateRating(\$orders, \$expert) {", "return \$rating / \$totalWeight;\n}\n?>");
                          if ($expert['redistribution_rate'] > 0.2) $totalWeight -= 1;
                          if ($expert['redistribution_rate'] > 0.1) $totalWeight -= 0.3;
                          if ($expert['late_delivery_rate'] > 0.2) $totalWeight -= 0.4;
                          if ($expert['politeness_positive_rate'] > 0.7) $totalWeight += 0.2;
                          // total rating
                          return $rating / $totalWeight;
                        }
                      }
